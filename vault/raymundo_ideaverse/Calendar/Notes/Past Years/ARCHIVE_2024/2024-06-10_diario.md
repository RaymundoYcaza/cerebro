let docs = dv.pages('"Atlas/Notes/X/Quotes"');

dv.list(randos);

function getRandos(list, max, itemNum) {
  var items = [];
  for (var i=0;i<itemNum;i++) {
    items.push(list[Math.floor(Math.random() * max)].file.link);
  }
  return items;
}
